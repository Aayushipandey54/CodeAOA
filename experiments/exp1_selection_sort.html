/* Selection Sort - O(n^2) time, O(1) space */
#include <stdio.h>

void selectionSort(int a[], int n){
    for(int i=0;i<n-1;i++){
        int minIdx = i;
        for(int j=i+1;j<n;j++)
            if(a[j] < a[minIdx]) minIdx = j;
        int t = a[i]; a[i] = a[minIdx]; a[minIdx] = t;
    }
}

int main(){
    int a[] = {64,25,12,22,11};
    int n = sizeof(a)/sizeof(a[0]);
    selectionSort(a,n);
    printf("Sorted: ");
    for(int i=0;i<n;i++) printf("%d ", a[i]);
    printf("\n");
    return 0;
}
/* Time Complexity: Best/Average/Worst = O(n^2). Space = O(1). */
