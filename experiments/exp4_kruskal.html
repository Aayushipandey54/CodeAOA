Kruskal’s Algorithm (Minimum Spanning Tree)

Code

/* Kruskal's MST using Union-Find.
   Time Complexity: O(E log E)
   Space: O(E + V)
*/
#include <stdio.h>
#include <stdlib.h>

typedef struct { int u,v,w; } Edge;
int cmp(const void *p,const void *q){ return ((Edge*)p)->w - ((Edge*)q)->w; }

int findp(int parent[], int x){
    if(parent[x]==x) return x;
    return parent[x]=findp(parent,parent[x]);
}
void unite(int parent[], int rnk[], int a, int b){
    a=findp(parent,a); b=findp(parent,b);
    if(a==b) return;
    if(rnk[a]<rnk[b]) parent[a]=b;
    else if(rnk[b]<rnk[a]) parent[b]=a;
    else { parent[b]=a; rnk[a]++; }
}

int main(){
    int V = 4;
    Edge edges[] = { {0,1,10}, {0,2,6}, {0,3,5}, {1,3,15}, {2,3,4} };
    int E = sizeof(edges)/sizeof(edges[0]);
    qsort(edges,E,sizeof(Edge),cmp);
    int parent[V], rnk[V]; for(int i=0;i<V;i++){ parent[i]=i; rnk[i]=0; }
    int mst_w = 0;
    printf("Kruskal edges selected:\n");
    for(int i=0;i<E;i++){
        int u = edges[i].u, v = edges[i].v;
        if(findp(parent,u)!=findp(parent,v)){
            unite(parent,rnk,u,v);
            printf("%d - %d : %d\n", u, v, edges[i].w);
            mst_w += edges[i].w;
        }
    }
    printf("Total MST weight = %d\n", mst_w);
    return 0;
}
/* Compare with Prim:
   - Prim: O(E log V)
   - Kruskal: O(E log E)
*/
