/* Travelling Salesperson Problem (TSP) using Dynamic Programming - Held-Karp Algorithm
   Time Complexity: O(n^2 * 2^n)
   Space Complexity: O(n * 2^n)
   Example: n = 4 cities
*/

#include <stdio.h>
#include <limits.h>

#define N 4
#define INF 999999

int min(int a, int b) { return (a < b) ? a : b; }

int main() {
    int cost[N][N] = {
        {INF, 20, 30, 10},
        {15, INF, 16, 4},
        {3, 5, INF, 2},
        {19, 6, 18, INF}
    };

    int FULL = 1 << N;
    int dp[FULL][N];

    for (int mask = 0; mask < FULL; mask++)
        for (int i = 0; i < N; i++)
            dp[mask][i] = INF;

    dp[1][0] = 0;  // Start at city 0

    for (int mask = 1; mask < FULL; mask++) {
        for (int u = 0; u < N; u++) {
            if (!(mask & (1 << u))) continue;
            for (int v = 0; v < N; v++) {
                if (mask & (1 << v)) continue;
                if (cost[u][v] == INF) continue;
                int nmask = mask | (1 << v);
                dp[nmask][v] = min(dp[nmask][v], dp[mask][u] + cost[u][v]);
            }
        }
    }

    int ans = INF;
    for (int i = 1; i < N; i++) {
        if (cost[i][0] != INF)
            ans = min(ans, dp[FULL - 1][i] + cost[i][0]);
    }

    printf("TSP Minimum Cost = %d\\n", ans);
    return 0;
}

/*
Example Output:
----------------
TSP Minimum Cost = 35
Path: 0 → 3 → 1 → 2 → 0
Total Cost = 10 + 4 + 16 + 5 = 35
*/
